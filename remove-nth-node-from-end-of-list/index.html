  func removeNthFromEnd(_ head: ListNode?, _ n: Int) -> ListNode? {
      var dummy = head
      var fast = head, slow = head
      for i in 0..<n {
          fast = fast?.next
          if fast == nil {
              return head?.next
          }
      }
      while fast?.next != nil {
          slow = slow?.next
          fast = fast?.next
      }
      slow?.next = slow?.next?.next
      return dummy
  }
