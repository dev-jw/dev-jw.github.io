int majorityElement(int *nums, int n) {
    int candidate = -1;
    int count = 0;
    for (int i = 0; i < n; i++) {
        int num = nums[i];
        if (num == candidate)
            ++count;
        else if (--count < 0) {
            int num = nums[i];
            candidate = num;
            count = 1;
        }

    return -1;
}
